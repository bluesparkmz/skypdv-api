Minimal Integration

React + FastAPI para auth BlueSpark MZ

Aqui tens o mínimo necessário para um app consumidor autenticar com a BlueSpark MZ usando:

  • login com conta BlueSpark no frontend React
  • login Google no frontend React
  • Google One Tap no frontend React
  • backend FastAPI para validar token, sincronizar user local e proteger rotas

Passo 2: frontend React mínimo

src/lib/authApi.ts
const API_URL = import.meta.env.VITE_API_URL;
const ACCOUNTS_URL = import.meta.env.VITE_BLUESPARK_ACCOUNTS_URL;
const PRODUCT_CODE = import.meta.env.VITE_BLUESPARK_PRODUCT_CODE || "skyvenda";

export async function loginWithBlueSpark(identifier, password) {
  const res = await fetch(`${API_URL}/user/token`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ identifier, password })
  });

  const data = await res.json();
  if (!res.ok) throw new Error(data.detail || "Falha no login");
  localStorage.setItem("auth_token", data.access_token);
  localStorage.setItem("refresh_token", data.refresh_token || "");
  return data;
}

export async function loginWithGoogleOneTap(credential) {
  const res = await fetch(`${API_URL}/user/auth/one-tap`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ credential })
  });

  const data = await res.json();
  if (!res.ok) throw new Error(data.detail || "Falha no Google One Tap");
  localStorage.setItem("auth_token", data.access_token);
  localStorage.setItem("refresh_token", data.refresh_token || "");
  return data;
}

export function buildGoogleRedirectUrl() {
  const next = encodeURIComponent(`${window.location.origin}/auth/success`);
  return `${ACCOUNTS_URL}/auth/google/start?product_code=${PRODUCT_CODE}&next=${next}`;
}

export function getAuthHeader() {
  const token = localStorage.getItem("auth_token");
  return token ? { Authorization: `Bearer ${token}` } : {};
}
src/components/LoginForm.jsx
import { useState } from "react";
import { loginWithBlueSpark, buildGoogleRedirectUrl } from "../lib/authApi";

export default function LoginForm() {
  const [identifier, setIdentifier] = useState("");
  const [password, setPassword] = useState("");
  const [error, setError] = useState("");
  const [loading, setLoading] = useState(false);

  async function onSubmit(e) {
    e.preventDefault();
    setError("");
    setLoading(true);
    try {
      await loginWithBlueSpark(identifier, password);
      window.location.href = "/dashboard";
    } catch (err) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  }

  return (
    <form onSubmit={onSubmit}>
      <input value={identifier} onChange={(e) => setIdentifier(e.target.value)} placeholder="Email ou username" />
      <input type="password" value={password} onChange={(e) => setPassword(e.target.value)} placeholder="Password" />
      <button disabled={loading} type="submit">{loading ? "Entrando..." : "Entrar com BlueSpark"}</button>
      <button type="button" onClick={() => (window.location.href = buildGoogleRedirectUrl())}>
        Entrar com Google
      </button>
      {error ? <p>{error}</p> : null}
    </form>
  );
}
src/components/GoogleOneTap.jsx
import { useEffect, useRef } from "react";
import { loginWithGoogleOneTap } from "../lib/authApi";

const GOOGLE_CLIENT_ID = import.meta.env.VITE_GOOGLE_CLIENT_ID;

export default function GoogleOneTap() {
  const initializedRef = useRef(false);

  useEffect(() => {
    if (!GOOGLE_CLIENT_ID) return;
    if (localStorage.getItem("auth_token")) return;

    const script = document.createElement("script");
    script.src = "https://accounts.google.com/gsi/client";
    script.async = true;
    script.defer = true;
    script.onload = init;
    document.body.appendChild(script);

    function init() {
      if (initializedRef.current) return;
      if (!window.google?.accounts?.id) return;

      initializedRef.current = true;
      window.google.accounts.id.initialize({
        client_id: GOOGLE_CLIENT_ID,
        callback: async (response) => {
          if (!response.credential) return;
          try {
            await loginWithGoogleOneTap(response.credential);
            window.location.href = "/dashboard";
          } catch (err) {
            console.error(err);
          }
        },
        auto_select: false,
        cancel_on_tap_outside: false,
        use_fedcm_for_prompt: true
      });

      window.google.accounts.id.prompt();
    }

    return () => {
      try { window.google?.accounts?.id?.cancel(); } catch {}
    };
  }, []);

  return null;
}

import { useEffect } from "react";

export default function AuthSuccess() {
  useEffect(() => {
    const params = new URLSearchParams(window.location.search);
    const accessToken = params.get("access_token");
    const refreshToken = params.get("refresh_token");

    if (accessToken) {
      localStorage.setItem("auth_token", accessToken);
      if (refreshToken) localStorage.setItem("refresh_token", refreshToken);
      window.location.href = "/dashboard";
    } else {
      window.location.href = "/login";
    }
  }, []);

  return <p>A concluir autenticação...</p>;
}

Passo 3: backend FastAPI mínimo

app/auth_client.py
import os
from typing import Any

import httpx
from fastapi import HTTPException, status

BLUESPARK_ACCOUNTS_URL = os.getenv("BLUESPARK_ACCOUNTS_URL", "https://accounts.bluesparkmz.com").rstrip("/")
BLUESPARK_PRODUCT_CODE = os.getenv("BLUESPARK_PRODUCT_CODE", "skyvenda")
TIMEOUT = float(os.getenv("BLUESPARK_AUTH_TIMEOUT_SECONDS", "15"))

def accounts_request(method: str, path: str, *, json: dict | None = None, token: str | None = None) -> dict[str, Any]:
    headers = {}
    if token:
        headers["Authorization"] = f"Bearer {token}"

    try:
        with httpx.Client(timeout=TIMEOUT) as client:
            response = client.request(method, f"{BLUESPARK_ACCOUNTS_URL}{path}", json=json, headers=headers)
    except httpx.HTTPError as exc:
        raise HTTPException(status_code=status.HTTP_502_BAD_GATEWAY, detail="BlueSpark indisponivel.") from exc

    data = response.json() if response.content else {}
    if response.status_code >= 400:
        raise HTTPException(status_code=response.status_code, detail=data.get("detail", "Falha na auth central."))
    return data

def login_with_central_auth(identifier: str, password: str) -> dict:
    return accounts_request("POST", "/auth/login", json={
        "identifier": identifier,
        "password": password,
        "product_code": BLUESPARK_PRODUCT_CODE,
    })

def google_one_tap_with_central_auth(credential: str) -> dict:
    return accounts_request("POST", "/auth/google/one-tap", json={
        "credential": credential,
        "product_code": BLUESPARK_PRODUCT_CODE,
    })

def introspect_central_token(token: str) -> dict:
    return accounts_request("POST", "/auth/introspect", json={
        "token": token,
        "product_code": BLUESPARK_PRODUCT_CODE,
    })

app/security.py
from fastapi import Depends, HTTPException, status
from fastapi.security import OAuth2PasswordBearer
from sqlalchemy.orm import Session

from .auth_client import introspect_central_token
from .database import get_db
from .models import User

oauth2_scheme = OAuth2PasswordBearer(tokenUrl="/user/token")

def sync_local_user_from_claims(db: Session, claims: dict) -> User:
    central_user_id = str(claims.get("central_user_id") or claims.get("sub") or "").strip()
    email = (claims.get("email") or "").strip().lower()
    username = (claims.get("username") or email.split("@")[0] or f"user{central_user_id}").strip()
    name = (claims.get("name") or claims.get("full_name") or username).strip()

    if not central_user_id:
        raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Token sem central_user_id.")
    if not email:
        raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Token sem email.")

    user = db.query(User).filter(User.central_user_id == central_user_id).first()
    if not user:
        user = db.query(User).filter(User.email == email).first()

    if user:
        user.central_user_id = central_user_id
        user.email = email
        if not user.name:
            user.name = name
        db.commit()
        db.refresh(user)
        return user

    user = User(central_user_id=central_user_id, email=email, username=username, name=name)
    db.add(user)
    db.commit()
    db.refresh(user)
    return user

def get_current_user(token: str = Depends(oauth2_scheme), db: Session = Depends(get_db)) -> User:
    claims = introspect_central_token(token)
    if not claims.get("active"):
        raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Sessao invalida.")
    return sync_local_user_from_claims(db, claims)
app/routes.py
from fastapi import APIRouter, Depends
from .auth_client import google_one_tap_with_central_auth, login_with_central_auth
from .schemas import GoogleOneTapRequest, LoginRequest
from .security import get_current_user

router = APIRouter()

@router.post("/user/token")
def login(payload: LoginRequest):
    return login_with_central_auth(payload.identifier, payload.password)

@router.post("/user/auth/one-tap")
def google_one_tap(payload: GoogleOneTapRequest):
    return google_one_tap_with_central_auth(payload.credential)

@router.get("/user/profile")
def profile(current_user = Depends(get_current_user)):
    return {
        "id": current_user.id,
        "central_user_id": current_user.central_user_id,
        "email": current_user.email,
        "username": current_user.username,
        "name": current_user.name,
    }

Passo 4: validação passo a passo

  1. O utilizador abre o frontend React em /login.
  2. Se usar senha, o React chama POST /user/token no backend do app.
  3. O backend FastAPI chama POST /auth/login na BlueSpark com product_code.
  4. A BlueSpark devolve access_token e refresh_token.
  5. O frontend guarda o token e passa a chamar as rotas protegidas do app.
  6. Ao abrir /user/profile, o backend lê o Bearer token.
  7. O backend faz /auth/introspect na BlueSpark.
  8. Se active=true, o backend procura o user local por central_user_id.
  9. Se o user local não existir, cria automaticamente.
  10. Se usar Google One Tap, o React recebe credential do Google e chama POST /user/auth/one-tap.
  11. O backend envia essa credential para /auth/google/one-tap da BlueSpark.
  12. Se usar Google redirect normal, o botão leva o utilizador para /auth/google/start da BlueSpark.
  13. Depois do callback central, a BlueSpark redireciona para /auth/success no app com os tokens.
  14. O frontend lê os tokens da query string, guarda e entra no app.

Passo 5: o que validar quando não funcionar

  • Confirmar que o product_code enviado é o do app certo.
  • Confirmar que o token vai no header Authorization: Bearer ....
  • Confirmar que a resposta de /auth/introspect vem com active=true.
  • Confirmar que o token contém sub, email e, idealmente, central_user_id.
  • Confirmar que o utilizador local está a ser criado ou atualizado no backend do app.
  • Confirmar que o frontend não continua a chamar rotas antigas de login local.
  • Confirmar que o Google Client ID é o mesmo da BlueSpark e que os origins estão autorizados.
Regra prática: o frontend pode iniciar a auth, mas a verdade da sessão fica sempre na BlueSpark e o backend do app só confia depois de validar o token com a auth central.
